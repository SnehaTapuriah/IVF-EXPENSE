Google Sheets Backend Setup (one-time)

All data is stored as one JSON blob in a Google Sheet, chunked across rows so there's no size limit. No CORS issues.

  1. Create a new Google Sheet.
  2. Open Extensions → Apps Script.
  3. Delete any existing code and paste the script below. Save (Ctrl+S).
  4. Click Deploy → New deployment → ⚙ → Web app.
  5. Set Execute as = Me, Who has access = Anyone. Click Deploy, authorise.
  6. Copy the Web app URL, paste it below, click Save & Connect.
function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; var data = sheet.getDataRange().getValues(); var json = ''; for (var i = 0; i < data.length; i++) json += data[i][0] || ''; return ContentService.createTextOutput(json || '{}') .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; var body = e.postData.contents; sheet.clear(); var CHUNK = 45000; var rows = []; for (var i = 0; i < body.length; i += CHUNK) { rows.push([body.substring(i, i + CHUNK)]); } if (rows.length === 0) rows.push(['']); sheet.getRange(1, 1, rows.length, 1).setValues(rows); return ContentService.createTextOutput(JSON.stringify({ok: true, chunks: rows.length, bytes: body.length})) .setMimeType(ContentService.MimeType.JSON); }
